@TableDecorator({ title: 'Product' })
export class ProductEntity extends BaseEntity {

  // Dropdown đơn giản — auto lookup /Product/lookup
  @DropDownDecorator({ label: 'Danh mục', required: true,
    lookup: LookupData.Reference(CategoryEntity) })
  CategoryId: string;

  // Multi-select
  @DropDownDecorator({ label: 'Tags', multiple: true,
    lookup: LookupData.Reference(TagEntity) })
  TagIds: string;

  // Với popup thêm mới
  @DropDownDecorator({
    label: 'Nhà cung cấp',
    lookup: LookupData.Reference(SupplierEntity),
    popup: PopupData.Auto()
  })
  SupplierId: string;
}
export class CategoryEntity extends BaseEntity {

  // Tree — danh mục phân cấp
  @DropDownDecorator({
    label: 'Danh mục cha',
    type: DropdownType.Tree,
    lookup: LookupData.Reference(CategoryEntity)
  })
  ParentId: string;

  // Grid — hiển thị dạng bảng
  @DropDownDecorator({
    label: 'Sản phẩm',
    type: DropdownType.Grid,
    lookup: LookupData.Reference(ProductEntity,
      ['Code', 'Name', 'Price'])
  })
  ProductId: string;
}
