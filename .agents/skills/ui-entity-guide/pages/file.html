@TableDecorator({ title: 'Contract' })
export class ContractEntity extends BaseEntity {

  // File PDF — bắt buộc
  @FileDecorator({
    label: 'Hợp đồng PDF',
    required: true,
    fileTypes: [FileTypeFormat.PDF]
  })
  ContractFile: string;

  // File văn phòng — PDF + Word + Excel
  @FileDecorator({
    label: 'Tài liệu',
    fileTypes: [FileTypeFormat.DOCUMENTS]
  })
  Document: string;
}
export class ProjectEntity extends BaseEntity {

  // Nhiều file — private, kéo thả
  @FileDecorator({
    label: 'Attachments',
    multiple: true,
    max: 10,
    dragable: true,
    private: true,
    size: 50,
    columnClass: ColumnClassFormat.Column12
  })
  Attachments: string;
}
