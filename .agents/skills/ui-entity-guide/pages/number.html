@TableDecorator({ title: 'Product' })
export class ProductEntity extends BaseEntity {

  // Giá sản phẩm — bắt buộc, cho search
  @NumberDecorator({ label: 'Giá', required: true,
    min: 0, max: 1000000000, allowSearch: true })
  Price: number;

  // Số lượng tồn kho
  @NumberDecorator({ label: 'Tồn kho', min: 0, max: 99999 })
  Stock: number;

  // Thứ tự hiển thị
  @NumberDecorator({ label: 'Thứ tự', type: NumberType.Numberic })
  Order: number;

  // Đánh giá sao
  @NumberDecorator({ type: NumberType.Star, max: 5 })
  Rating: number;
}
export class InvoiceEntity extends BaseEntity {

  // Số thập phân — step=0.01
  @NumberDecorator({
    label: 'Chiết khấu %',
    min: 0, max: 100,
    step: 0.01, decimals: 2
  })
  Discount: number;

  // Range slider — phần trăm
  @NumberDecorator({
    type: NumberType.Range,
    min: 0, max: 100, step: 5
  })
  Progress: number;

  // Cho phép số âm (lợi nhuận/lỗ)
  @NumberDecorator({
    label: 'Lợi nhuận',
    allowNegative: true
  })
  Profit: number;
}
