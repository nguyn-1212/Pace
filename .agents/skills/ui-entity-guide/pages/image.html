@TableDecorator({ title: 'Product' })
export class ProductEntity extends BaseEntity {

  // Ảnh đại diện — bắt buộc
  @ImageDecorator({ label: 'Ảnh', required: true })
  Image: string;

  // Ảnh private — max 5MB
  @ImageDecorator({
    label: 'CMND',
    size: 5,
    private: true
  })
  IdCard: string;
}
export class GalleryEntity extends BaseEntity {

  // Nhiều ảnh — kéo thả sắp xếp
  @ImageDecorator({
    label: 'Bộ sưu tập',
    multiple: true,
    max: 20,
    dragable: true,
    note: true,
    columnClass: ColumnClassFormat.Column12
  })
  Images: string;
}
