⚠️ @DateTimeDecorator() không truyền options → mặc định DateTimeType.DateTime, format=dd/mm/yyyy hh:ii, icon=la-calendar.
DateTimeType.Date
Chỉ chọn ngày. Format mặc định: dd/mm/yyyy.
Ngày sinh, ngày tạo, deadline
DateTimeType.DateTime
Chọn ngày + giờ. Format: dd/mm/yyyy hh:ii.
Mặc định — thời gian chính xác
DateTimeType.Time
Chỉ chọn giờ/phút. Format: hh:ii.
Giờ bắt đầu, giờ kết thúc
DateTimeType.Month
Chọn tháng/năm. Format: mm/yyyy.
Báo cáo theo tháng
DateTimeType.Year
Chỉ chọn năm. Format: yyyy.
Năm tài chính, năm học
DateTimeType.DayMonth
Chọn ngày/tháng (không năm). Format: dd/mm.
Ngày lễ, sinh nhật hàng năm
DateTimeType.DateRange
Chọn khoảng ngày (từ — đến). Dùng maxRange.
Kỳ báo cáo, thời gian áp dụng

@TableDecorator({ title: 'Task' })
export class TaskEntity extends BaseEntity {

  // Date — không cho chọn quá khứ
  @DateTimeDecorator({ label: 'Deadline', type: DateTimeType.Date, minCurent: true })
  Deadline: Date;

  // DateTime mặc định
  @DateTimeDecorator({ label: 'Ngày tạo', readonly: true })
  CreatedDate: Date;

  // Chỉ chọn giờ — bước 15 phút
  @DateTimeDecorator({
    label: 'Giờ bắt đầu', type: DateTimeType.Time,
    time: { minuteStep: 15, minHours: 8, maxHours: 18 }
  })
  StartTime: Date;

  // StartDate → EndDate phụ thuộc
  @DateTimeDecorator({ label: 'Từ ngày', type: DateTimeType.Date })
  StartDate: Date;

  @DateTimeDecorator({ label: 'Đến ngày', type: DateTimeType.Date,
    minDepend: 'StartDate' })
  EndDate: Date;
}
export class ReportEntity extends BaseEntity {

  // DateRange — tối đa 90 ngày
  @DateTimeDecorator({
    label: 'Kỳ báo cáo',
    type: DateTimeType.DateRange,
    maxRange: 90,
    columnClass: ColumnClassFormat.Column6
  })
  ReportPeriod: Date;

  // Month picker
  @DateTimeDecorator({
    label: 'Tháng', type: DateTimeType.Month,
    required: true,
    columnClass: ColumnClassFormat.Column3
  })
  ReportMonth: Date;

  // Year picker
  @DateTimeDecorator({
    label: 'Năm', type: DateTimeType.Year,
    columnClass: ColumnClassFormat.Column3
  })
  ReportYear: Date;
}
