2b. Config trong OnModelCreating
modelBuilder.Entity<Product>(entity =>
{
    entity.ToTable("product");       // tên bảng lowercase
    entity.HasKey(e => e.Id);
    entity.HasIndex(e => e.Name);
    entity.HasIndex(e => e.Code);
    entity.Property(e => e.Name).IsRequired().HasMaxLength(250);
    entity.Property(e => e.Code).HasMaxLength(50);
    entity.Property(e => e.Description).HasMaxLength(1000);

    // FK (nếu có)
    entity.HasOne(d => d.Category)
        .WithMany(p => p.Products)
        .HasForeignKey(d => d.CategoryId)
        .HasConstraintName("FK_Products_CategoryId");

    // ⚠️ Luôn thêm 2 dòng này cuối cùng
    entity.HasOne(d => d.CreatedByUser).WithMany().HasForeignKey(c => c.CreatedBy);
    entity.HasOne(d => d.UpdatedByUser).WithMany().HasForeignKey(c => c.UpdatedBy);
});
⚠️Quy tắc bắt buộc: ToTable tên bảng lowercase, luôn HasKey(Id), HasMaxLength cho mọi string, constraint đặt tên FK_[Table]_[Column], 2 dòng CreatedByUser/UpdatedByUser ở cuối.

⚡ Nâng cao — Custom endpoint & Service

🔧 Thêm endpoint custom

Vẫn kế thừa AdminBaseController<T>, thêm action riêng. Dùng Repository (protected).

public class ProductController : AdminBaseController<Product>
{
    public ProductController(IServiceProvider sp) : base(sp) { }

    // GET /api/admin/Product/Lookup
    [HttpGet("Lookup")]
    [Authorize(AuthenticationSchemes = "Bearer")]
    public IActionResult Lookup()
    {
        var result = Repository.Queryable()
            .Where(c => c.IsActive == true)
            .Select(c => new { c.Id, c.Name, c.Code })
            .ToList();
        return Ok(ResultApi.ToEntity(result));
    }
}
📊 Override /Items (custom select + join)
[HttpPost("Items")]
[Authorize(AuthenticationSchemes = "Bearer")]
public async Task<IActionResult> GetAllAsync([FromBody] TableData obj)
{
    var result = await Repository.Queryable().AsNoTracking()
        .Select(c => new
        {
            c.Id, c.Name, c.Code,
            c.IsActive, c.IsDelete,
            c.CreatedDate, c.UpdatedDate,
            // Join navigation
            CategoryName = c.Category != null ? c.Category.Name : "",
            CreatedBy = c.CreatedByUser != null ? c.CreatedByUser.UserName : "",
        })
        .ToQueryAsync(obj); // paging+filter+sort tự động
    return Ok(result);
}
⚠️ToQueryAsync(obj) tự áp dụng paging, filter, sort từ TableData (do Grid gửi lên). Chỉ cần Select dữ liệu, phần còn lại tự xử lý.
