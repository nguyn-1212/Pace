@TableDecorator({ title: 'Contract' })
export class ContractEntity extends BaseEntity {

  // Chữ ký — tất cả phương thức
  @SignatureDecorator({ label: 'Chữ ký', required: true })
  Signature: string;

  // Chỉ vẽ tay
  @SignatureDecorator({
    label: 'Chữ ký tay',
    type: SignatureType.Draw,
    required: true
  })
  HandSignature: string;
}
export class ApprovalEntity extends BaseEntity {

  // Upload ảnh private — max 2MB
  @SignatureDecorator({
    label: 'Chữ ký xác nhận',
    type: SignatureType.Image,
    size: 2,
    private: true,
    store: StoreType.DatabaseBase64
  })
  ApprovalSign: string;

  // Text — chữ ký nhanh
  @SignatureDecorator({
    type: SignatureType.Text,
    columnClass: ColumnClassFormat.Column6
  })
  QuickSign: string;
}
