@TableDecorator({ title: 'Course' })
export class CourseEntity extends BaseEntity {

  // Video giới thiệu — 1 video, max 100MB
  @VideoDecorator({ label: 'Video giới thiệu' })
  IntroVideo: string;

  // Video bắt buộc — max 50MB
  @VideoDecorator({
    label: 'Video bài giảng',
    required: true,
    size: 50
  })
  LectureVideo: string;
}
export class LessonEntity extends BaseEntity {

  // Nhiều video — kéo thả
  @VideoDecorator({
    label: 'Videos',
    multiple: true,
    max: 5,
    dragable: true,
    columnClass: ColumnClassFormat.Column12
  })
  Videos: string;
}
