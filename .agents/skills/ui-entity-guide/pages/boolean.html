@TableDecorator({ title: 'Product' })
export class ProductEntity extends BaseEntity {

  // Checkbox mặc định
  @BooleanDecorator({ label: 'Hoạt động' })
  IsActive: boolean;

  // Toggle switch
  @BooleanDecorator({ label: 'Hiển thị', type: BooleanType.Toggle })
  IsVisible: boolean;

  // Có filter trên grid
  @BooleanDecorator({ label: 'Nổi bật', allowSearch: true })
  IsFeatured: boolean;
}
export class SettingEntity extends BaseEntity {

  // Toggle — cho phép bỏ chọn → null
  @BooleanDecorator({
    label: 'Sao lưu tự động',
    type: BooleanType.Toggle,
    allowUncheck: true,
    columnClass: ColumnClassFormat.Column6
  })
  BackupEnabled: boolean;

  // Radio button — Có/Không
  @BooleanDecorator({
    label: 'Gửi báo cáo tuần',
    type: BooleanType.RadioButton,
    columnClass: ColumnClassFormat.Column6
  })
  WeeklyReport: boolean;
}
