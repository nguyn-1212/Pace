@TableDecorator({ title: 'Product' })
export class ProductEntity extends BaseEntity {

  // Text thường — bắt buộc, cho search
  @StringDecorator({ label: 'Tên', required: true, allowSearch: true, type: StringType.Text, max: 250 })
  Name: string;

  // Code — unique + uppercase
  @StringDecorator({ required: true, type: StringType.Code, max: 20,
    unique: LookupUniqueData.Reference(ProductEntity, 'Code'),
    textTransform: TextTransformType.UpperCase })
  Code: string;

  // Textarea
  @StringDecorator({ type: StringType.MultiText, rows: 5 })
  Description: string;

  // Color picker
  @StringDecorator({ label: 'Màu sắc', type: StringType.Color })
  Color: string;
}
export class UserEntity extends BaseEntity {

  // Email — unique
  @StringDecorator({ required: true, type: StringType.Email, max: 80,
    unique: LookupUniqueData.Reference(UserEntity, 'Email') })
  Email: string;

  // Phone plain text
  @StringDecorator({ type: StringType.PhoneText, min: 10, max: 10 })
  Phone: string;

  // Password + xác nhận
  @StringDecorator({ required: true, type: StringType.Password })
  Password: string;

  @StringDecorator({ required: true, type: StringType.Password,
    requiredMatch: 'Password' })
  ConfirmPassword: string;
}
export class EmailTemplateEntity extends BaseEntity {

  // HTML editor với inject biến
  @StringDecorator({
    required: true, type: StringType.Html, max: 2000000,
    variables: [
      { name: 'FullName' }, { name: 'Link' }, { name: 'Email' }
    ]
  })
  Content: string;

  @StringDecorator({ type: StringType.Json })
  Settings: string;

  @StringDecorator({ type: StringType.Tag, maxTags: 20 })
  Keywords: string;
}
